package com.irurueta.hermes;

/**
 * Contains information about found changes in a collection item.
 */
public class CollectionItemChange {

    /**
     * Action that has been performed on the item, such as being added, removed, changed or moved within a collection.
     */
    private final CollectionItemChangeAction action;

    /**
     * Constructor.
     *
     * @param listItemChangeAction action that has been performed on the item, such as being added, removed, changed or
     *                             moved within a list.
     * @throws IllegalArgumentException if provided action is null.
     */
    public CollectionItemChange(final CollectionItemChangeAction listItemChangeAction) {
        if (listItemChangeAction == null) {
            throw new IllegalArgumentException();
        }

        this.action = listItemChangeAction;
    }

    /**
     * Gets the action that has been performed on the item, such as being added, removed, changed or moved within a
     * collection.
     * @return the action that has been performed on the item.
     */
    public CollectionItemChangeAction getAction() {
        return action;
    }
}

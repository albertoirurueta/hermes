17  package com.irurueta.hermes;
18  
19  import java.util.ArrayList;
20  import java.util.List;
21  
22  /**
23   * Abstract class to detect changes between two lists of items.
24   * Notice that returned changes in this class indicate the positions of the items when each change is applied
25   * sequentially to the old list.
26   *
27   * @param <T> type of items.
28   */
29  public abstract class AbstractSequentialListItemChangeDetector<T> {
30  
31      /**
32       * Detects changes between two lists of items.
33       *
34       * @param newItems new items.
35       * @param oldItems old items.
36       * @return a list of found changes.
37       */
38      public List<ListItemChange> detectChanges(final List<T> newItems, final List<T> oldItems) {
39          if (oldItems == null || newItems == null) {
40              throw new IllegalArgumentException();
41          }
42  
43          final var changes = new ArrayList<ListItemChange>();
44  
45          final var newCopy = new ArrayList<>(newItems);
46          final var oldCopy = new ArrayList<>(oldItems);
47  
48          // removes
49          changes.addAll(buildRemoves(newCopy, oldCopy));
50  
51          // inserts
52          changes.addAll(buildInserts(newCopy, oldCopy));
53  
54          // moves
55          changes.addAll(buildMoves(newCopy, oldCopy));
56  
57          // changed items
58          changes.addAll(buildChanges(newCopy, oldCopy));
59  
60          return changes;
61      }
62  
63      /**
64       * Determines whether the content of an item has changed.
65       *
66       * @param newItem new item.
67       * @param oldItem old item.
68       * @return true if the content of the item has changed, false otherwise.
69       */
70      protected abstract boolean hasContentChanged(final T newItem, final T oldItem);
71  
72      /**
73       * Determines whether an item is not contained in a list.
74       *
75       * @param items list of items.
76       * @param item item to be checked.
77       * @return true if the item is not contained in the list, false otherwise.
78       */
79      protected abstract boolean notContains(final List<T> items, final T item);
80  
81      /**
82       * Determines the index of an item in a list.
83       *
84       * @param items list of items.
85       * @param item item to be checked.
86       * @return index of the item in the list, or -1 if the item is not contained in the list.
87       */
88      protected abstract int indexOf(final List<T> items, final T item);
89  
90      /**
91       * Finds removed items and build a list of detected remove changes.
92       * Notice that this method modifies the old items copy to avoid checking removed items again.
93       *
94       * @param newItemsCopy new items copy.
95       * @param oldItemsCopy old items copy.
96       * @return a list of detected remove changes.
97       */
98      private List<RemovedListItemChange<T>> buildRemoves(final List<T> newItemsCopy, final List<T> oldItemsCopy) {
99          final var removes = new ArrayList<RemovedListItemChange<T>>();
100 
101         for (var i = 0; i < oldItemsCopy.size(); i++) {
102             final var oldItem = oldItemsCopy.get(i);
103             if (notContains(newItemsCopy, oldItem)) {
104                 // item has been removed on the new list, we remove it to avoid checking it again
105                 oldItemsCopy.remove(i);
106                 removes.add(new RemovedListItemChange<>(oldItem, i));
107                 // decrease i so that next iteration checks the same position again
108                 i--;
109             }
110         }
111         return removes;
112     }
113 
114     /**
115      * Finds inserted items and build a list of detected insert changes.
116      * Notice that this method modifies the old items copy to avoid checking inserted items again.
117      *
118      * @param newItemsCopy new items copy.
119      * @param oldItemsCopy old items copy.
120      * @return a list of detected insert changes.
121      */
122     private List<InsertedListItemChange<T>> buildInserts(final List<T> newItemsCopy, final List<T> oldItemsCopy) {
123         final var inserts = new ArrayList<InsertedListItemChange<T>>();
124 
125         final var newSize = newItemsCopy.size();
126         for (var i = 0; i < newSize; i++) {
127             final var newItem = newItemsCopy.get(i);
128             if (notContains(oldItemsCopy, newItem)) {
129                 // this item has been added on the new list
130                 final var pos = Math.min(i, oldItemsCopy.size());
131                 inserts.add(new InsertedListItemChange<>(newItem, pos));
132 
133                 // make old items resemble new items so that changes and moves are later detected at proper positions
134                 if (i >= oldItemsCopy.size()) {
135                     oldItemsCopy.add(newItem);
136                 } else {
137                     oldItemsCopy.add(i, newItem);
138                 }
139             }
140         }
141 
142         return inserts;
143     }
144 
145     /**
146      * Finds moved items and build a list of detected move changes.
147      * Notice that this method modifies the old items copy to avoid checking moved items again.
148      *
149      * @param newItemsCopy new items copy.
150      * @param oldItemsCopy old items copy.
151      * @return a list of detected move changes.
152      */
153     private List<MovedListItemChange<T>> buildMoves(final List<T> newItemsCopy, final List<T> oldItemsCopy) {
154         final var moves = new ArrayList<MovedListItemChange<T>>();
155 
156         final var itemsToMove = new ArrayList<>(oldItemsCopy);
157         for (var i = 0; i < itemsToMove.size(); i++) {
158             final var item = itemsToMove.get(i);
159 
160             final var pos1 = indexOf(oldItemsCopy, item);
161             final var pos2 = indexOf(newItemsCopy, item);
162 
163             if (pos2 >= 0 && pos2 < itemsToMove.size() && pos1 != pos2) {
164                 // item has been moved
165                 final var oldItem = oldItemsCopy.get(pos1);
166                 final var newItem = newItemsCopy.get(pos2);
167                 moves.add(new MovedListItemChange<>(oldItem, newItem, pos1, pos2));
168                 // remove item from old items to avoid checking it again
169                 final var removedItem = oldItemsCopy.remove(pos1);
170                 if (pos2 >= oldItemsCopy.size()) {
171                     oldItemsCopy.add(removedItem);
172                 } else {
173                     oldItemsCopy.add(pos2, removedItem);
174                 }
175             }
176         }
177 
178         return moves;
179     }
180 
181     /**
182      * Finds changed items and build a list of detected update changes.
183      *
184      * @param newItemsCopy new items copy.
185      * @param oldItemsCopy old items copy.
186      * @return a list of detected update changes.
187      */
188     private List<UpdatedListItemChange<T>> buildChanges(final List<T> newItemsCopy, final List<T> oldItemsCopy) {
189         final var changes = new ArrayList<UpdatedListItemChange<T>>();
190 
191         var pos = 0;
192         for (final var newItem : newItemsCopy) {
193             final var oldPos = indexOf(oldItemsCopy, newItem);
194             if (oldPos >= 0) {
195                 final var oldItem = oldItemsCopy.get(oldPos);
196 
197                 // compare both items
198                 if (hasContentChanged(newItem, oldItem)) {
199                     // item has changed
200                     changes.add(new UpdatedListItemChange<>(oldItem, newItem, pos));
201                 }
202             }
203             pos++;
204         }
205 
206         return changes;
207     }
208 }
